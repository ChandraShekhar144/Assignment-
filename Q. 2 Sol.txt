public class AnagramCheck {

    // Method to check if two strings are anagrams
    public static boolean areAnagrams(String str1, String str2) {
        // If the lengths of the strings are different, they can't be anagrams
        if (str1.length() != str2.length()) {
            return false;
        }

        // Convert both strings to lowercase to make the check case-insensitive
        str1 = str1.toLowerCase();
        str2 = str2.toLowerCase();

        // Create an array to count the frequency of each character in both strings
        int[] charCount = new int[26]; // Assuming only lowercase English letters

        // Increment the count for each character in the first string
        for (int i = 0; i < str1.length(); i++) {
            charCount[str1.charAt(i) - 'a']++;
        }

        // Decrement the count for each character in the second string
        for (int i = 0; i < str2.length(); i++) {
            charCount[str2.charAt(i) - 'a']--;
        }

        // If any count is non-zero, the strings are not anagrams
        for (int count : charCount) {
            if (count != 0) {
                return false;
            }
        }

        return true;
    }

    public static void main(String[] args) {
        // Example 1
        String str1 = "listen";
        String str2 = "silent";

        // Check if the strings are anagrams and display the result for Example 1
        if (areAnagrams(str1, str2)) {
            System.out.println("Input: \"" + str1 + "\", \"" + str2 + "\"");
            System.out.println("Output: true");
        } else {
            System.out.println("Input: \"" + str1 + "\", \"" + str2 + "\"");
            System.out.println("Output: false");
        }

        // Example 2
        str1 = "hello";
        str2 = "world";

        // Check if the strings are anagrams and display the result for Example 2
        if (areAnagrams(str1, str2)) {
            System.out.println("Input: \"" + str1 + "\", \"" + str2 + "\"");
            System.out.println("Output: true");
        } else {
            System.out.println("Input: \"" + str1 + "\", \"" + str2 + "\"");
            System.out.println("Output: false");
        }
    }
}
